## Skeleton

Demo:

```tsx
import React from 'react';
import { Skeleton } from '@obiusm/react-components';

export default () => (
  <Skeleton isVisible={true}>
    <div style={{ width: '100px', height: '100px' }}></div>
  </Skeleton>
);
```
